213 #define _adc_isr_full_code(adcp) { \
214  if ((adcp)->grpp->circular) { \
215  /* Callback handling.*/ \
216  if ((adcp)->grpp->end_cb != NULL) { \
217  if ((adcp)->depth > 1) { \
218  /* Invokes the callback passing the 2nd half of the buffer.*/ \
219  size_t half = (adcp)->depth / 2; \
220  size_t half_index = half * (adcp)->grpp->num_channels; \
221  (adcp)->grpp->end_cb(adcp, (adcp)->samples + half_index, half); \
222  } \
223  else { \
224  /* Invokes the callback passing the whole buffer.*/ \
225  (adcp)->grpp->end_cb(adcp, (adcp)->samples, (adcp)->depth); \
226  } \
227  } \
228  } \
229  else { \
230  /* End conversion.*/ \
231  adc_lld_stop_conversion(adcp); \
232  if ((adcp)->grpp->end_cb != NULL) { \
233  (adcp)->state = ADC_COMPLETE; \
234  /* Invoke the callback passing the whole buffer.*/ \
235  (adcp)->grpp->end_cb(adcp, (adcp)->samples, (adcp)->depth); \
236  if ((adcp)->state == ADC_COMPLETE) { \
237  (adcp)->state = ADC_READY; \
238  (adcp)->grpp = NULL; \
239  } \
240  } \
241  else { \
242  (adcp)->state = ADC_READY; \
243  (adcp)->grpp = NULL; \
244  } \
245  _adc_wakeup_isr(adcp); \
246  } \
247 }

264 #define _adc_isr_error_code(adcp, err) { \
265  adc_lld_stop_conversion(adcp); \
266  if ((adcp)->grpp->error_cb != NULL) { \
267  (adcp)->state = ADC_ERROR; \
268  (adcp)->grpp->error_cb(adcp, err); \
269  if ((adcp)->state == ADC_ERROR) \

272  (adcp)->grpp = NULL; \

274 }
