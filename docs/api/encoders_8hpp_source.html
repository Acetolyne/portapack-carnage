40  struct encoder_def_t {
41  char name[16]; // Encoder chip ref/name
42  char address_symbols[8]; // List of possible symbols like "01", "01F"...
43  char data_symbols[8]; // Same
44  uint16_t clk_per_symbol; // Oscillator periods per symbol
45  uint16_t clk_per_fragment; // Oscillator periods per symbol fragment (state)
46  char bit_format[4][20]; // List of fragments for each symbol in previous *_symbols list order
47  uint8_t word_length; // Total # of symbols (not counting sync)
48  char word_format[32]; // A for Address, D for Data, S for sync
49  char sync[64]; // Like bit_format
50  uint32_t default_speed; // Default encoder clk frequency (often set by shitty resistor)
51  uint8_t repeat_min; // Minimum repeat count
52  uint16_t pause_symbols; // Length of pause between repeats in symbols
53  };
