131 inline float sin_f32(const float w) {
132  constexpr float normalize = 1.0 / (2 * pi);
133 
134  const float x = w * normalize;
135  const int32_t x_int = std::floor(x);
136  const float x_frac = x - x_int;
137 
138  const float n = x_frac * sine_table_f32_period;
139  const int32_t n_int = static_cast<uint32_t>(n) & sine_table_f32_index_mask;
140  const float n_frac = n - n_int;
141 
142  const float p0 = sine_table_f32[n_int + 0];
143  const float p1 = sine_table_f32[n_int + 1];
144  const float diff = p1 - p0;
145  const float result = p0 + n_frac * diff;
146  return result;
147 }
