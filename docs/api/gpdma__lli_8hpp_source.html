203  void set_lli_sequential(ChainType chain_type) {
204  for(auto& item : *chain) {
205  item.lli = lli_pointer(&item + 1);
206  }
207  if( chain_type == ChainType::Loop ) {
208  chain[chain->size() - 1].lli = lli_pointer(&chain[0]);
209  } else {
210  chain[chain->size() - 1].lli = lli_pointer(nullptr);
211  }
212  }
