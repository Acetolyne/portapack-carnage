30 struct I2CClockConfig {
31  float clock_source_f;
32  float bus_f;
33  float high_period_ns;
34 
35  static constexpr float period_ns(const float f) {
36  return 1e9 / f;
37  }
38 
39  constexpr uint32_t i2c_period_count() const {
40  return period_ns(bus_f) / period_ns(clock_source_f) + 0.5f;
41  }
42 
43  constexpr uint32_t i2c_high_count() const {
44  return high_period_ns / period_ns(clock_source_f) + 0.5f;
45  }
46 
47  constexpr uint32_t i2c_low_count() const {
48  return i2c_period_count() - i2c_high_count();
49  }
50 };
