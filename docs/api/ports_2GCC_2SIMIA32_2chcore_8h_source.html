119 #define SETUP_CONTEXT(workspace, wsize, pf, arg) { \
120  uint8_t *esp = (uint8_t *)workspace + wsize; \
121  APUSH(esp, 0); \
122  uint8_t *savebp = esp; \
123  AALIGN(esp, 15, 8); \
124  APUSH(esp, arg); \
125  APUSH(esp, pf); \
126  APUSH(esp, 0); \
127  esp -= sizeof(struct intctx); \
