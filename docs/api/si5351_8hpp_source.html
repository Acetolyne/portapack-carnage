293  constexpr uint32_t p2() const {
294  return 128 * b - c * (uint32_t)(128 * (float)b / (float)c);

298  return c;

305  constexpr MultisynthFractionalReg reg(const uint8_t multisynth_n) const {
306  return {
307  uint8_t(42 + (multisynth_n * 8)),
