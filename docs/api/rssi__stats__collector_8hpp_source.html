31 class RSSIStatisticsCollector {
32 public:
33  template<typename Callback>
34  void process(const rf::rssi::buffer_t& buffer, Callback callback) {
35  auto p = buffer.p;
36  if( p == nullptr ) {
37  return;
38  }
39 
40  if( statistics.count == 0 ) {
41  const auto value_0 = *p;
42  statistics.min = value_0;
43  statistics.max = value_0;
44  }
45 
46  const auto end = &p[buffer.count];
47  while(p < end) {
48  const uint32_t value = *(p++);
49 
50  if( statistics.min > value ) {
51  statistics.min = value;
52  }
53  if( statistics.max < value ) {
54  statistics.max = value;
55  }
56 
57  statistics.accumulator += value;
58  }
59  statistics.count += buffer.count;
60 
61  const size_t samples_per_update = buffer.sampling_rate * update_interval;
62 
63  if( statistics.count >= samples_per_update ) {
64  callback(statistics);
65  statistics.accumulator = 0;
66  statistics.count = 0;
67  }
68  }
69 
70 private:
71  static constexpr float update_interval { 0.1f };
72  RSSIStatistics statistics { };
73 };
