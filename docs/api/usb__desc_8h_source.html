51 struct usb_conf_desc_header {
52  struct usb_desc_head h; /* Size 0x09, Id 0x02 */
53  uint16_t tot_leng; /* Total length of data */
54  uint8_t num_int; /* Number of interfaces */
55  uint8_t conf_val; /* Configuration selector */
56  uint8_t conf_desc; /* Index of conf string desc */
57  uint8_t attr; /* Attribute bitmap:
58  * 7 : Bus powered

77 struct usb_ep_desc {

79  uint8_t ep_addr; /* Endpoint address:
