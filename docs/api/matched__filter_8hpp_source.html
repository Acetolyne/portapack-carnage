32 // This filter contains "magic" (optimizations) that expect the taps to
33 // combine a low-pass filter with a complex sinusoid that performs shifting of
34 // the input signal to 0Hz/DC. This also means that the taps length must be
35 // a multiple of the complex sinusoid period.
36 
37 class MatchedFilter {
38 public:
39  using sample_t = std::complex<float>;
40  using tap_t = std::complex<float>;
41 
42  using taps_t = tap_t[];
43 
44  template<class T>
45  MatchedFilter(
46  const T& taps,
47  size_t decimation_factor = 1
48  ) {
49  configure(taps, decimation_factor);
50  }
51 
52  template<class T>
53  void configure(
54  const T& taps,
55  size_t decimation_factor
56  ) {
57  configure(taps.data(), taps.size(), decimation_factor);
58  }
59 
60  bool execute_once(const sample_t input);
61 
62  float get_output() const {
63  return output;
64  }
65 
66 private:
67  using samples_t = sample_t[];
68 
69  std::unique_ptr<samples_t> samples_ { };
70  std::unique_ptr<taps_t> taps_reversed_ { };
71  size_t taps_count_ { 0 };
72  size_t decimation_factor_ { 1 };
73  size_t decimation_phase { 0 };
74  float output { 0 };
75 
76  void shift_by_decimation_factor();
77 
78  void advance_decimation_phase() {
79  decimation_phase = (decimation_phase + 1) % decimation_factor_;
80  }
81 
82  bool is_new_decimation_cycle() const {
83  return (decimation_phase == 0);
84  }
85 
86  void configure(
87  const tap_t* const taps,
88  const size_t taps_count,
89  const size_t decimation_factor
90  );
91 };
