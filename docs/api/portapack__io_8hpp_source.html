184  size_t word_count = byte_count / 2;
185  while(word_count) {
186  const auto word = lcd_read_data();
187  *(byte++) = word >> 8;

299  void lcd_command(const uint32_t value) {
300  data_write_high(0); /* Drive high byte (with zero -- don't care) */
301  dir_write(); /* Turn around data bus, MCU->CPLD */
302  addr(0); /* Indicate command */
303  __asm__("nop");
304  __asm__("nop");
305  __asm__("nop");
306  lcd_wr_assert(); /* Latch high byte */
307 
308  data_write_low(value); /* Drive low byte (pass-through) */
309  __asm__("nop");
310  __asm__("nop");
311  __asm__("nop");
312  lcd_wr_deassert(); /* Complete write operation */
313 
314  addr(1); /* Set up for data phase (most likely after a command) */
315  }

317  void lcd_write_data(const uint32_t value) __attribute__((always_inline)) {
318  // NOTE: Assumes and DIR=0 and ADDR=1 from command phase.
319  data_write_high(value); /* Drive high byte */
320  __asm__("nop");
321  lcd_wr_assert(); /* Latch high byte */
322 
323  data_write_low(value); /* Drive low byte (pass-through) */
324  __asm__("nop");
325  __asm__("nop");
326  __asm__("nop");
327  lcd_wr_deassert(); /* Complete write operation */
328  }

330  uint32_t lcd_read_data() {
331  // NOTE: Assumes ADDR=1 from command phase.
332  dir_read();
333 
334  /* Start read operation */
335  lcd_rd_assert();
336  /* Wait for passthrough data(15:8) to settle -- ~16ns (3 cycles) typical */
337  /* Wait for read control L duration (355ns) */
338  halPolledDelay(71); // 355ns
339  const auto value_high = data_read();
340 
341  /* Latch data[7:0] */
342  lcd_rd_deassert();
343  /* Wait for latched data[7:0] to settle -- ~26ns (5 cycles) typical */
344  /* Wait for read control H duration (90ns) */
345  halPolledDelay(18); // 90ns
346 
347  const auto value_low = data_read();
348  return (value_high << 8) | value_low;
349  }
