132 struct RegistersWidgetConfig {
133  size_t registers_count;
134  size_t register_bits;
135 
136  constexpr size_t legend_length() const {
137  return (registers_count >= 0x10) ? 2 : 1;
138  }
139 
140  constexpr size_t legend_width() const {
141  return legend_length() * 8;
142  }
143 
144  constexpr size_t value_length() const {
145  return (register_bits + 3) / 4;
146  }
147 
148  constexpr size_t value_width() const {
149  return value_length() * 8;
150  }
151 
152  constexpr size_t registers_per_row() const {
153  return (value_length() >= 3) ? 4 : 8;
154  }
155 
156  constexpr size_t registers_row_length() const {
157  return (registers_per_row() * (value_length() + 1)) - 1;
158  }
159 
160  constexpr size_t registers_row_width() const {
161  return registers_row_length() * 8;
162  }
163 
164  constexpr size_t row_width() const {
165  return legend_width() + 8 + registers_row_width();
166  }
167 
168  constexpr size_t rows() const {
169  return registers_count / registers_per_row();
170  }
171 };
