65 /*
66  * I/O ports initial setup, this configuration is established soon after reset
67  * in the initialization code.
68  * Please refer to the STM32 Reference Manual for details.
69  */
70 #define PIN_MODE_INPUT(n) (0 << ((n) * 2))
71 #define PIN_MODE_OUTPUT(n) (1 << ((n) * 2))
72 #define PIN_MODE_ALTERNATE(n) (2 << ((n) * 2))
73 #define PIN_MODE_ANALOG(n) (3 << ((n) * 2))
74 #define PIN_OTYPE_PUSHPULL(n) (0 << (n))
75 #define PIN_OTYPE_OPENDRAIN(n) (1 << (n))
76 #define PIN_OSPEED_2M(n) (0 << ((n) * 2))
77 #define PIN_OSPEED_25M(n) (1 << ((n) * 2))
78 #define PIN_OSPEED_50M(n) (2 << ((n) * 2))
79 #define PIN_OSPEED_100M(n) (3 << ((n) * 2))
80 #define PIN_PUDR_FLOATING(n) (0 << ((n) * 2))
81 #define PIN_PUDR_PULLUP(n) (1 << ((n) * 2))
82 #define PIN_PUDR_PULLDOWN(n) (2 << ((n) * 2))
83 #define PIN_AFIO_AF(n, v) ((v##U) << ((n % 8) * 4))
