46 // From http://www.graphics.stanford.edu/~seander/bithacks.html, nice !
47 constexpr inline uint8_t sstv_parity(uint8_t code) {
48  uint8_t out = code;
49  out ^= code >> 4;
50  out &= 0x0F;
51  return (((0b0110100110010110 >> out) & 1) << 7) | code;
52 }
