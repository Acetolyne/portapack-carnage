49  /* The "start_bit" winds up being the MSB of the returned field value. */
50  /* The BitRemap functor determines which bits are read from the source
51  * packet. */
52  uint32_t read(const size_t start_bit, const size_t length) const {
53  uint32_t value = 0;
54  for(size_t i=start_bit; i<(start_bit + length); i++) {
55  value = (value << 1) | data[bit_remap(i)];
56  }
57  return value;
58  }
59 
60 private:
61  const T& data;
62  const BitRemap bit_remap { };
63 };
