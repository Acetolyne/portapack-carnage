511  __STATIC_INLINE uint32_t arm_recip_q31(
512  q31_t in,
513  q31_t * dst,
514  q31_t * pRecipTable)
515  {
516 
517  uint32_t out, tempVal;
518  uint32_t index, i;
519  uint32_t signBits;
520 
521  if(in > 0)
522  {
523  signBits = __CLZ(in) - 1;
524  }
525  else
526  {
527  signBits = __CLZ(-in) - 1;
528  }
529 
530  /* Convert input sample to 1.31 format */
531  in = in << signBits;
532 
533  /* calculation of index for initial approximated Val */
534  index = (uint32_t) (in >> 24u);
535  index = (index & INDEX_MASK);
536 
537  /* 1.31 with exp 1 */
538  out = pRecipTable[index];
539 
540  /* calculation of reciprocal value */
541  /* running approximation for two iterations */
542  for (i = 0u; i < 2u; i++)
543  {
544  tempVal = (q31_t) (((q63_t) in * out) >> 31u);
545  tempVal = 0x7FFFFFFF - tempVal;
546  /* 1.31 with exp 1 */
547  //out = (q31_t) (((q63_t) out * tempVal) >> 30u);
548  out = (q31_t) clip_q63_to_q31(((q63_t) out * tempVal) >> 30u);
549  }
550 
551  /* write output */
552  *dst = out;
553 
554  /* return num of signbits of out = 1/in value */
555  return (signBits + 1u);
556 
557  }
