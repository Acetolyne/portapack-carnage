38  /* https://github.com/AE9RB/fftbench/blob/master/cxlr.hpp
39  * Nice trick from AE9RB (David Turnbull) to get compiler to produce simpler
40  * fma (fused multiply-accumulate) instead of worrying about NaN handling
41  */
42  inline complex<float>
43  operator*(const complex<float>& v1, const complex<float>& v2) {
44  return complex<float> {
45  v1.real() * v2.real() - v1.imag() * v2.imag(),
46  v1.real() * v2.imag() + v1.imag() * v2.real()
47  };
48  }

51 template<typename T, size_t N>
52 void fft_swap(const buffer_c16_t src, std::array<T, N>& dst) {
53  static_assert(power_of_two(N), "only defined for N == power of two");
54 
55  for(size_t i=0; i<N; i++) {
56  const size_t i_rev = __RBIT(i) >> (32 - log_2(N));
57  const auto s = src.p[i];
58  dst[i_rev] = {
59  static_cast<typename T::value_type>(s.real()),
60  static_cast<typename T::value_type>(s.imag())
61  };
62  }
63 }

102 template<typename T, size_t N>
103 void fft_c_preswapped(std::array<T, N>& data, const size_t from, const size_t to) {
104  static_assert(power_of_two(N), "only defined for N == power of two");
105  constexpr auto K = log_2(N);
106  if ((to > K) || (from > K)) return;
107 
108  constexpr size_t K_max = 8;
109  static_assert(K <= K_max, "No FFT twiddle factors for K > 8");
110  static constexpr std::array<std::complex<float>, K_max> wp_table { {
111  { -2.0f, 0.0f }, // 2
112  { -1.0f, -1.0f }, // 4
113  { -0.2928932188134524756f, -0.7071067811865475244f }, // 8
114  { -0.076120467488713243872f, -0.38268343236508977173f }, // 16
115  { -0.019214719596769550874f, -0.19509032201612826785f }, // 32
116  { -0.0048152733278031137552f, -0.098017140329560601994f }, // 64
117  { -0.0012045437948276072852f, -0.049067674327418014255f }, // 128
118  { -0.00030118130379577988423f, -0.024541228522912288032f }, // 256
119  } };
120 
121  /* Provide data to this function, pre-swapped. */
122  for(size_t k = from; k < to; k++) {
123  const size_t mmax = 1 << k;
124  const auto wp = wp_table[k];
125  T w { 1.0f, 0.0f };
126  for(size_t m = 0; m < mmax; ++m) {
127  for(size_t i = m; i < N; i += mmax * 2) {
128  const size_t j = i + mmax;
129  const T temp = w * data[j];
130  data[j] = data[i] - temp;
131  data[i] += temp;
132  }
133  w += w * wp;
134  }
135  }
136 }
