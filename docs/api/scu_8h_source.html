417 /*
418 * SCU PIN Normal Drive:
419 * The configuration registers for normal-drive pins control the following pins:
420 * - P0_0 and P0_1
421 * - P1_0 to P1_16 and P1_18 to P1_20
422 * - P2_0 to P2_2 and P2_6 to P2_13
423 * - P3_0 to P3_2 and P3_4 to P3_8
424 * - P4_0 to P4_10
425 * - P5_0 to P5_7
426 * - P6_0 to P6_12
427 * - P7_0 to P7_7
428 * - P8_3 to P8_8
429 * - P9_0 to P9_6
430 * - PA_0 and PA_4
431 * - PB_0 to PB_6
432 * - PC_0 to PC_14
433 * - PE_0 to PE_15
434 * - PF_0 to PF_11
435 *
436 * Pin configuration registers for High-Drive pins.
437 * The configuration registers for high-drive pins control the following pins:
438 * - P1_17
439 * - P2_3 to P2_5
440 * - P8_0 to P8_2
441 * - PA_1 to PA_3
442 *
443 * Pin configuration registers for High-Speed pins.
444 * This register controls the following pins:
445 * - P3_3 and pins CLK0 to CLK3.
446 */
447 typedef enum {
