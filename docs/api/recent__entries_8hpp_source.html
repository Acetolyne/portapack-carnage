39 template<typename ContainerType, typename Key>
40 typename ContainerType::const_iterator find(const ContainerType& entries, const Key key) {
41  return std::find_if(
42  std::begin(entries), std::end(entries),
43  [key](typename ContainerType::const_reference e) { return e.key() == key; }
44  );
45 }

47 template<typename ContainerType>
48 static void truncate_entries(ContainerType& entries, const size_t entries_max = 64) {
49  while(entries.size() > entries_max) {

54 template<typename ContainerType, typename Key>
55 typename ContainerType::reference on_packet(ContainerType& entries, const Key key) {
56  auto matching_recent = find(entries, key);
57  if( matching_recent != std::end(entries) ) {
58  // Found within. Move to front of list, increment counter.
59  entries.push_front(*matching_recent);
60  entries.erase(matching_recent);
61  } else {
62  entries.emplace_front(key);
63  truncate_entries(entries);
64  }
65 
66  return entries.front();
67 }

69 template<typename ContainerType>
70 static std::pair<typename ContainerType::const_iterator, typename ContainerType::const_iterator> range_around(
71  const ContainerType& entries,
72  typename ContainerType::const_iterator item,
73  const size_t count
74 ) {
75  auto start = item;
76  auto end = item;
77  size_t i = 0;
78 
79  // Move start iterator toward first entry.
80  while( (start != std::begin(entries)) && (i < count / 2) ) {
81  std::advance(start, -1);
82  i++;
83  }
84 
85  // Move end iterator toward last entry.
86  while( (end != std::end(entries)) && (i < count) ) {
87  std::advance(end, 1);
88  i++;
89  }
90 
91  return { start, end };
92 }

126 class RecentEntriesTable : public Widget {
127 public:
128  using Entry = typename Entries::value_type;

139  void paint(Painter& painter) override {
140  const auto r = screen_rect();
141  const auto& s = style();
142 
143  Rect target_rect { r.location(), { r.width(), s.font.line_height() }};
144  const size_t visible_item_count = r.height() / s.font.line_height();
145 
146  auto selected = find(recent, selected_key);
147  if( selected == std::end(recent) ) {
148  selected = std::begin(recent);
149  }
150 
151  auto range = range_around(recent, selected, visible_item_count);
152 
153  for(auto p = range.first; p != range.second; p++) {
154  const auto& entry = *p;
155  const auto is_selected_key = (selected_key == entry.key());
156  const auto item_style = (has_focus() && is_selected_key) ? s.invert() : s;
157  draw(entry, target_rect, painter, item_style);
158  target_rect += { 0, target_rect.height() };
159  }
160 
161  painter.fill_rectangle(
162  { target_rect.left(), target_rect.top(), target_rect.width(), r.bottom() - target_rect.top() },
163  style().background
164  );
165  }
166 
167  bool on_encoder(const EncoderEvent event) override {
168  advance(event);
169  return true;
170  }
171 
172  bool on_key(const ui::KeyEvent event) override {
173  if( event == ui::KeyEvent::Select ) {
174  if( on_select ) {
175  const auto selected = find(recent, selected_key);
176  if( selected != std::end(recent) ) {
177  on_select(*selected);
178  return true;
179  }
180  }
181  }
182  return false;
183  }
184 
185  void on_focus() override {
186  advance(0);
187  }
188 
189 private:
190  Entries& recent;
191 
192  using EntryKey = typename Entry::Key;
193  EntryKey selected_key = Entry::invalid_key;
194 
195  void advance(const int32_t amount) {
196  auto selected = find(recent, selected_key);
197  if( selected == std::end(recent) ) {
198  if( recent.empty() ) {
199  selected_key = Entry::invalid_key;
200  } else {
201  selected_key = recent.front().key();
202  }
203  } else {
204  if( amount < 0 ) {
205  if( selected != std::begin(recent) ) {
206  std::advance(selected, -1);
207  }
208  }
209  if( amount > 0 ) {
210  std::advance(selected, 1);
211  if( selected == std::end(recent) ) {
212  return;
213  }
214  }
215  selected_key = selected->key();
216  }
217 
218  set_dirty();
219  }

222  const Entry& entry,
223  const Rect& target_rect,
224  Painter& painter,
225  const Style& style
226  );
227 };
