103  bool push(const void* const buf, const size_t len) {
104  chMtxLock(&mutex_write);
105  const auto result = fifo.in_r(buf, len);
106  chMtxUnlock();
107 
108  const bool success = (result == len);
109  if( success ) {
110  signal();
111  }
