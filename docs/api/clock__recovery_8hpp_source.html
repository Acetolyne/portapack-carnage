33 class GardnerTimingErrorDetector {
34 public:
35  static constexpr size_t samples_per_symbol { 2 };
36 
37  /*
38  Expects retimed samples at a rate of twice the expected symbol rate.
39  Calculates timing error, sends symbol and error to handler.
40  */
41  template<typename SymbolHandler>
42  void operator()(
43  const float in,
44  SymbolHandler symbol_handler
45  ) {
46  /* NOTE: Algorithm is sensitive to input magnitude. Timing error value
47  * will scale proportionally. Best practice is to use error sign only.
48  */
49  t[2] = t[1];
50  t[1] = t[0];
51  t[0] = in;
52 
53  if( symbol_phase == 0 ) {
54  const auto symbol = t[0];
55  const float lateness = (t[0] - t[2]) * t[1];
56  symbol_handler(symbol, lateness);
57  }
58 
59  symbol_phase = (symbol_phase + 1) % samples_per_symbol;
60  }
61 
62 private:
63  std::array<float, 3> t { };
64  size_t symbol_phase { 0 };
65 };

67 class LinearErrorFilter {
68 public:
69  LinearErrorFilter(
70  const float filter_alpha = 0.95f,
71  const float error_weight = -1.0f
72  ) : filter_alpha { filter_alpha },
73  error_weight { error_weight }
74  {
75  }
76 
77  float operator()(
78  const float error
79  ) {
80  error_filtered = filter_alpha * error_filtered + (1.0f - filter_alpha) * error;
81  return error_filtered * error_weight;
82  }
83 
84 private:
85  const float filter_alpha;
86  const float error_weight;
87  float error_filtered { 0.0f };
88 };

170  void symbol_callback(const float symbol, const float lateness) {
171  // NOTE: This check is to avoid std::function nullptr check, which
172  // brings in "_ZSt25__throw_bad_function_callv" and a lot of extra code.
173  // TODO: Make symbol_handler known at compile time.
174  if( symbol_handler) {
175  symbol_handler(symbol);
176  }
177 
178  const float adjustment = error_filter(lateness);
179  resampler.advance(adjustment);
180  }
