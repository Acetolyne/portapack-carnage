125  */
126 
127  constexpr Calibration(
128  const std::array<DigitizerPoint, 3>& s,
129  const std::array<ui::Point, 3>& d
130  ) : k { (s[0].x - s[2].x) * (s[1].y - s[2].y) - (s[1].x - s[2].x) * (s[0].y - s[2].y) },
131  a { (d[0].x() - d[2].x()) * (s[1].y - s[2].y) - (d[1].x() - d[2].x()) * (s[0].y - s[2].y) },
132  b { (s[0].x - s[2].x) * (d[1].x() - d[2].x()) - (d[0].x() - d[2].x()) * (s[1].x - s[2].x) },
133  c { s[0].y * (s[2].x * d[1].x() - s[1].x * d[2].x()) + s[1].y * (s[0].x * d[2].x() - s[2].x * d[0].x()) + s[2].y * (s[1].x * d[0].x() - s[0].x * d[1].x()) },
134  d { (d[0].y() - d[2].y()) * (s[1].y - s[2].y) - (d[1].y() - d[2].y()) * (s[0].y - s[2].y) },
135  e { (s[0].x - s[2].x) * (d[1].y() - d[2].y()) - (d[0].y() - d[2].y()) * (s[1].x - s[2].x) },

154 template<size_t N>
155 class Filter {
156 public:
157  constexpr Filter() = default;
158 
159  void reset() {
160  history.fill(0);
161  history_history = 0;
162  accumulator = 0;
163  n = 0;
164  }
165 
166  void feed(const sample_t value) {
167  accumulator = accumulator + value - history[n];
168  history[n] = value;
169  n = (n + 1) % history.size();
170 
171  history_history = (history_history << 1) | 1U;
172  }
173 
174  int32_t value() const {
175  return accumulator / N;
176  }
177 
178  bool stable(const uint32_t bound) const {
179  if( history_valid() ) {
180  const auto minmax = std::minmax_element(history.cbegin(), history.cend());
181  const auto min = *minmax.first;
182  const auto max = *minmax.second;
183  const uint32_t delta = max - min;
184  return (delta < bound);
185  } else {
186  return false;
187  }
188  }
189 
190 private:
191  static constexpr uint32_t history_history_mask { (1U << N) - 1 };
192 
193  std::array<sample_t, N> history { };
194  uint32_t history_history { 0 };
195  int32_t accumulator { 0 };
196  size_t n { 0 };
197 
198  bool history_valid() const {
199  return (history_history & history_history_mask) == history_history_mask;
200  }
201 };
